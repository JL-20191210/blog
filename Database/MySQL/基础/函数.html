еҮҪж•°

1. еӯ—з¬ҰдёІеҮҪж•°

иҝҷдәӣеҮҪж•°з”ЁдәҺж“ҚдҪңеӯ—з¬ҰдёІж•°жҚ®зұ»еһӢгҖӮ

  • CONCAT(str1, str2, ...)
    иҝһжҺҘеӨҡдёӘеӯ—з¬ҰдёІгҖӮ

    SELECT CONCAT('Hello', ' ', 'World');

  • LENGTH(str)
    иҝ”еӣһеӯ—з¬ҰдёІзҡ„еӯ—иҠӮй•ҝеәҰгҖӮ

    SELECT LENGTH('Hello');

  • SUBSTRING(str, start, length)
    жҸҗеҸ–еӯ—з¬ҰдёІзҡ„дёҖйғЁеҲҶгҖӮ

    SELECT SUBSTRING('Hello', 1, 3);  -- иҫ“еҮә 'Hel'

  • UPPER(str)
    е°Ҷеӯ—з¬ҰдёІиҪ¬жҚўдёәеӨ§еҶҷгҖӮ

    SELECT UPPER('hello');

  • LOWER(str)
    е°Ҷеӯ—з¬ҰдёІиҪ¬жҚўдёәе°ҸеҶҷгҖӮ

    SELECT LOWER('HELLO');

  • TRIM(str)
    еҲ йҷӨеӯ—з¬ҰдёІдёӨз«Ҝзҡ„з©әж јгҖӮ

    SELECT TRIM('   Hello   ');

  • REPLACE(str, old_substring, new_substring)
    жӣҝжҚўеӯ—з¬ҰдёІдёӯзҡ„еӯҗеӯ—з¬ҰдёІгҖӮ

    SELECT REPLACE('Hello World', 'World', 'MySQL');

  • LPAD(str, length, pad_string)
    е°Ҷеӯ—з¬ҰдёІеЎ«е……еҲ°жҢҮе®ҡй•ҝеәҰпјҢеЎ«е……еҸ‘з”ҹеңЁеӯ—з¬ҰдёІзҡ„е·Ұдҫ§гҖӮ

    SELECT LPAD('123', 5, '0');  -- иҫ“еҮә '00123'

  • RPAD(str, length, pad_string)
    е°Ҷеӯ—з¬ҰдёІеЎ«е……еҲ°жҢҮе®ҡй•ҝеәҰпјҢеЎ«е……еҸ‘з”ҹеңЁеӯ—з¬ҰдёІзҡ„еҸідҫ§гҖӮ

    SELECT RPAD('123', 5, '0');  -- иҫ“еҮә '12300'

2. ж•°еӯҰеҮҪж•°

иҝҷдәӣеҮҪж•°з”ЁдәҺиҝӣиЎҢж•°еӯҰи®Ўз®—гҖӮ

  • ABS(x)
    иҝ”еӣһ x зҡ„з»қеҜ№еҖјгҖӮ

    SELECT ABS(-5);  -- иҫ“еҮә 5

  • ROUND(x, d)
    е°Ҷж•°еӯ— x еӣӣиҲҚдә”е…ҘеҲ° d дҪҚе°Ҹж•°гҖӮ

    SELECT ROUND(123.456, 2);  -- иҫ“еҮә 123.46

  • FLOOR(x)
    иҝ”еӣһе°ҸдәҺжҲ–зӯүдәҺ x зҡ„жңҖеӨ§ж•ҙж•°гҖӮ

    SELECT FLOOR(5.8);  -- иҫ“еҮә 5

  • CEIL(x)
    иҝ”еӣһеӨ§дәҺжҲ–зӯүдәҺ x зҡ„жңҖе°Ҹж•ҙж•°гҖӮ

    SELECT CEIL(5.1);  -- иҫ“еҮә 6

  • POW(x, y)
    иҝ”еӣһ x зҡ„ y ж¬Ўж–№гҖӮ

    SELECT POW(2, 3);  -- иҫ“еҮә 8

  • RAND()
    иҝ”еӣһдёҖдёӘ 0 еҲ° 1 д№Ӣй—ҙзҡ„йҡҸжңәжө®еҠЁеҖјгҖӮ

    SELECT RAND();

3. ж—Ҙжңҹе’Ңж—¶й—ҙеҮҪж•°

иҝҷдәӣеҮҪж•°з”ЁдәҺеӨ„зҗҶж—Ҙжңҹе’Ңж—¶й—ҙж•°жҚ®зұ»еһӢгҖӮ

  • CURDATE()
    иҝ”еӣһеҪ“еүҚж—ҘжңҹгҖӮ

    SELECT CURDATE();  -- иҫ“еҮә '2025-02-03'

  • NOW()
    иҝ”еӣһеҪ“еүҚзҡ„ж—Ҙжңҹе’Ңж—¶й—ҙгҖӮ

    SELECT NOW();  -- иҫ“еҮә '2025-02-03 07:23:13'

  • DATE_ADD(date, INTERVAL value unit)
    еҗ‘ж—Ҙжңҹж·»еҠ жҢҮе®ҡзҡ„ж—¶й—ҙй—ҙйҡ”гҖӮ

    SELECT DATE_ADD('2025-02-03', INTERVAL 1 DAY);  -- иҫ“еҮә '2025-02-04'

  • DATE_SUB(date, INTERVAL value unit)
    д»Һж—ҘжңҹдёӯеҮҸеҺ»жҢҮе®ҡзҡ„ж—¶й—ҙй—ҙйҡ”гҖӮ

    SELECT DATE_SUB('2025-02-03', INTERVAL 1 MONTH);  -- иҫ“еҮә '2025-01-03'

  • DATEDIFF(date1, date2)
    иҝ”еӣһдёӨдёӘж—Ҙжңҹд№Ӣй—ҙзҡ„еӨ©ж•°е·®гҖӮ

    SELECT DATEDIFF('2025-02-03', '2025-01-01');  -- иҫ“еҮә 33

  • YEAR(date)
    иҝ”еӣһжҢҮе®ҡж—Ҙжңҹзҡ„е№ҙд»ҪйғЁеҲҶгҖӮ

    SELECT YEAR('2025-02-03');  -- иҫ“еҮә 2025

  • MONTH(date)
    иҝ”еӣһжҢҮе®ҡж—Ҙжңҹзҡ„жңҲд»ҪйғЁеҲҶгҖӮ

    SELECT MONTH('2025-02-03');  -- иҫ“еҮә 2

  • DAY(date)
    иҝ”еӣһжҢҮе®ҡж—Ҙжңҹзҡ„еӨ©йғЁеҲҶгҖӮ

    SELECT DAY('2025-02-03');  -- иҫ“еҮә 3

4. иҒҡеҗҲеҮҪж•°

иҒҡеҗҲеҮҪж•°з”ЁдәҺеӨ„зҗҶж•°жҚ®йӣҶе№¶иҝ”еӣһеҚ•дёҖзҡ„и®Ўз®—з»“жһңгҖӮ

  • COUNT(*)
    иҝ”еӣһиЎҢж•°гҖӮ

    SELECT COUNT(*) FROM employees;

  • SUM(expression)
    иҝ”еӣһиЎЁиҫҫејҸзҡ„жҖ»е’ҢгҖӮ

    SELECT SUM(salary) FROM employees;

  • AVG(expression)
    иҝ”еӣһиЎЁиҫҫејҸзҡ„е№іеқҮеҖјгҖӮ

    SELECT AVG(salary) FROM employees;

  • MAX(expression)
    иҝ”еӣһиЎЁиҫҫејҸзҡ„жңҖеӨ§еҖјгҖӮ

    SELECT MAX(salary) FROM employees;

  • MIN(expression)
    иҝ”еӣһиЎЁиҫҫејҸзҡ„жңҖе°ҸеҖјгҖӮ

    SELECT MIN(salary) FROM employees;

5. жқЎд»¶еҮҪж•°

з”ЁдәҺеҹәдәҺжқЎд»¶иҝӣиЎҢж•°жҚ®еӨ„зҗҶгҖӮ

  • IF(condition, true_value, false_value)
    еҰӮжһңжқЎд»¶дёәзңҹиҝ”еӣһ true_valueпјҢеҗҰеҲҷиҝ”еӣһ false_valueгҖӮ

    SELECT IF(salary > 5000, 'High', 'Low') FROM employees;

  • CASE WHEN
    е®һзҺ°жӣҙеӨҚжқӮзҡ„жқЎд»¶йҖ»иҫ‘гҖӮ

    SELECT 
  CASE 
    WHEN salary > 5000 THEN 'High'
    WHEN salary BETWEEN 3000 AND 5000 THEN 'Medium'
    ELSE 'Low'
  END AS salary_level
FROM employees;

  • IFNULL(expression, value)
    еҰӮжһңиЎЁиҫҫејҸзҡ„з»“жһңдёә NULLпјҢеҲҷиҝ”еӣһжҢҮе®ҡзҡ„ valueгҖӮ

    SELECT IFNULL(salary, 0) FROM employees;

6. зұ»еһӢиҪ¬жҚўеҮҪж•°

з”ЁдәҺиҪ¬жҚўж•°жҚ®зұ»еһӢгҖӮ

  • CAST(expression AS type)
    е°ҶиЎЁиҫҫејҸиҪ¬жҚўдёәжҢҮе®ҡзҡ„ж•°жҚ®зұ»еһӢгҖӮ

    SELECT CAST(123 AS CHAR);  -- иҫ“еҮә '123'

  • CONVERT(expression, type)
    е°ҶиЎЁиҫҫејҸиҪ¬жҚўдёәжҢҮе®ҡзҡ„ж•°жҚ®зұ»еһӢпјҲзұ»дјј CASTпјүгҖӮ

    SELECT CONVERT(123, CHAR);  -- иҫ“еҮә '123'

7. дҝЎжҒҜеҮҪж•°

з”ЁдәҺиҺ·еҸ–жңүе…іж•°жҚ®еә“жҲ–иЎЁзҡ„дҝЎжҒҜгҖӮ

  • DATABASE()
    иҝ”еӣһеҪ“еүҚж•°жҚ®еә“зҡ„еҗҚз§°гҖӮ

    SELECT DATABASE();

  • USER()
    иҝ”еӣһеҪ“еүҚз”ЁжҲ·зҡ„з”ЁжҲ·еҗҚгҖӮ

    SELECT USER();

  • VERSION()
    иҝ”еӣһ MySQL зҡ„зүҲжң¬еҸ·гҖӮ

    SELECT VERSION();
